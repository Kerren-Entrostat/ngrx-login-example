Dispatch

When we want to perform an action we "dispatch" it.

constructor(
    private httpClient: HttpClient,
    private store$: Store<AppState>
) {}

login(email: string, password: string) {
    this.httpClient.post('/login', { email, password })
        .subscribe(
            user => this.store$.dispatch(
                new LoginSuccessAction(user)
            ),
            error => this.store$.dispatch(
                new LoginFailureAction(error)
            ),
        );
}
